    $.getJSON( "/archive/concert/data.json", {

        var str = "";

                str += "<li><p>" + concert.month + "月 : <a href='/archive/concert/?id=" + concert.id + "'>" + concert.name + "</a><p></li>";
            }
            str += "</ul>";
        }
        //str += "</ul>";

        $('#holder').append(str);
    });
})();
</script>
